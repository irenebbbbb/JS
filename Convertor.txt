let value = prompt('Вкажіть значення, яке Ви хочете перерахувати в іншу одиницю виміру');
let unitOfMeasurement = prompt('Вкажіть одиницю виміру (км/год/кг)');
switch (unitOfMeasurement) {
    case 'км':
        console.log(value + ' км відповідає ' + value * 1000 + ' м');
        break;
    case 'год':
        console.log(value + ' год відповідає ' + value * 60 + ' хв.');
        break;
    case 'кг':
        console.log(value + ' кг відповідає ' + value * 1000 + ' г');
        break;
    default:
        console.log('Така одиниця виміру не підтримується');
}
